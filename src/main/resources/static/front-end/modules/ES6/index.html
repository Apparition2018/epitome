// 在http服务器上访问，否则将会有一个CORS保护报错 /***** index.html *****/ import operation from './math.js'; let product = operation().multiply(9, 9); $('.es6').html("9 * 9 = " + product); /***** math.js *****/ export default function operation() { var plus = function (a, b) { return a + b; }; var subtract = function (a, b) { return a - b; }; var multiply = function (a, b) { return a * b; }; var divide = function (a, b) { return a / b; }; return { plus: plus, subtract: subtract, multiply: multiply, divide: divide }; };
