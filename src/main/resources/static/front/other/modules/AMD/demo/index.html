RequireJS API
/***** page.js *****/ require.config({ paths: { 'jquery': 'https://cdn.bootcdn.net/ajax/libs/jquery/1.12.4/jquery.min.js' } }); require(['jquery', 'math'], function($, math) { var product = math.multiply(9, 9); $(".require").html("9 * 9 = " + product); }); /***** math.js *****/ define(function () { var plus = function (a, b) { return a + b; }; var subtract = function (a, b) { return a - b; }; var multiply = function (a, b) { return a * b; }; var divide = function (a, b) { return a / b; }; return { plus: plus, subtract: subtract, multiply: multiply, divide: divide } });
