/********** SQL Server **********/
-- CEILING
SELECT CEILING(3.44) -- 4
SELECT CEILING(3.46) -- 4

-- ISNUMERIC
SELECT ISNUMERIC(2.2) -- 1
SELECT ISNUMERIC('2') -- 1
SELECT ISNUMERIC('a') -- 0

-- RAND
SELECT RAND() -- 0 ~ 1 之间随机浮点值
GO
