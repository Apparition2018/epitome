>>opt                           Optional.ofNullable(expr)
>>par                           (expr)

>>sout                          System.out.println(expr)
>>stream                        Arrays.stream(expr)                                             ???
>>switch                        switch (expr)
>>synchronized                  synchronized (expr)
>>throw                         throw expr
>>try                           try { expr } catch (Exeception e)

>>var                           T name = expr
>>while                         while (expr) {}
